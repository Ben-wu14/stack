model = Sequential()

model.add(LSTM(
    input_dim=1,
    output_dim=seqence_length,
    return_sequences=True
))
model.add(Dropout(0.2))

model.add(LSTM(
    256,
    return_sequences=True
))
model.add(Dropout(0.2))

model.add(LSTM(
    128,
    return_sequences=True
))
model.add(Dropout(0.2))

model.add(LSTM(
    64,
    return_sequences=False
))
model.add(Dense(
    output_dim=1
))

model.add(Activation('linear'))

start = time.time()
model.compile(loss='mse',optimizer='rmsprop')
print('Compilation time:',time.time()-start)
